function numberOfPairs(gloves) {
  let sum=0;
  let pairSum = 0;
  let arr = gloves.join(',').split(',');
  let first
  while (arr.length>0){
    first = arr[0];
 arr.shift();
  if(arr.includes(first) === true) {
   sum++
    }
 while (arr.includes(first) === true) {
   sum++;
   arr.splice(arr.indexOf(first),1)
 }
    pairSum += Math.floor((sum)/2);

}
